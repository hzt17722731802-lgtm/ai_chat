## 员工管理接口

### 基础信息
- **接口前缀**: `/admin/employee`
- **实体**: `Employee`

### 1. 分页查询员工

#### 请求
- **URL**: `/admin/employee/page`
- **Method**: `GET`
- **参数**:
  - `pageNum`: 页码（默认：1）
  - `pageSize`: 每页数量（默认：10）
  - `name`: 员工姓名（可选）
  - `department`: 所属科室（可选）
  - `status`: 状态（可选）

#### 响应
```json
{
  "code": 200,
  "message": "操作成功",
  "data": {
    "records": [
      {
        "id": "123456",
        "employeeNo": "EMP001",
        "name": "张三",
        "department": "内科",
        "position": "主治医师",
        "phone": "13800138001",
        "email": "zhangsan@example.com",
        "performanceScore": 95.5,
        "salesAmount": 150000.00,
        "completedOrders": 120,
        "status": "ACTIVE",
        "createTime": "2024-01-15 10:30:00",
        "updateTime": "2024-03-20 15:45:00"
      }
    ],
    "total": 100,
    "size": 10,
    "current": 1,
    "pages": 10
  }
}
```

#### 前端调用示例
```javascript
// 分页查询员工
const fetchEmployees = async (pageNum = 1, pageSize = 10, params = {}) => {
  const url = new URL(`${BASE_URL}/admin/employee/page`);
  url.searchParams.append('pageNum', pageNum);
  url.searchParams.append('pageSize', pageSize);
  
  if (params.name) url.searchParams.append('name', params.name);
  if (params.department) url.searchParams.append('department', params.department);
  if (params.status) url.searchParams.append('status', params.status);
  
  const response = await fetch(url, {
    method: 'GET',
    headers: {
      'Authorization': 'Bearer your_token'
    }
  });
  
  const result = await response.json();
  return result;
};

// 调用示例
fetchEmployees(1, 10, { name: '张三', department: '内科', status: 'ACTIVE' });
```

- **请求参数**:

  - `salesAmount`: 销售额（可选）

```

```

```json
{
  "code": 200,
  "message": "业绩更新成功",
  "data": null
}
```

#### 前端调用示例
```javascript
const updatePerformance = async (id, performanceData) => {
  const url = new URL(`${BASE_URL}/admin/employee/${id}/performance`);
  
  if (performanceData.performanceScore) {
    url.searchParams.append('performanceScore', performanceData.performanceScore);
  }
  if (performanceData.salesAmount) {
    url.searchParams.append('salesAmount', performanceData.salesAmount);
  }
  if (performanceData.completedOrders) {
    url.searchParams.append('completedOrders', performanceData.completedOrders);
  }
  
  const response = await fetch(url, {
    method: 'PUT',
    headers: {
      'Authorization': 'Bearer your_token'
    }
  });
  
  const result = await response.json();
  return result;
};

// 调用示例
updatePerformance('123456', {
  performanceScore: 98.5,
  salesAmount: 200000.00,
  completedOrders: 150
});
```

|--------|---------|------|
| 0 | 禁用 | 用户被禁用，无法登录 |
| 1 | 启用 | 用户正常启用 |

| 3 | 已注销 | 用户已注销账号 |

### 员工状态（Employee.status）
| 状态值 | 状态名称 | 说明 |
|--------|---------|------|
| ACTIVE | 在职 | 员工正常在职 |
| INACTIVE | 离职 | 员工已离职 |
| ON_LEAVE | 休假 | 员工处于休假状态 |

### Neo4j 关系类型
| 关系类型 | 说明 | 示例 |
|---------|------|------|
| `recommand_drug` | 推荐药物 | 糖尿病 - recommand_drug -> 二甲双胍 |
| `common_drug` | 常用药物 | 高血压 - common_drug -> 阿司匹林 |
| `has_symptom` | 拥有症状 | 糖尿病 - has_symptom -> 多尿 |
| `belongs_to` | 所属科室 | 糖尿病 - belongs_to -> 内分泌科 |
| `need_check` | 需要检查 | 糖尿病 - need_check -> 空腹血糖 |
| `acompany_with` | 并发疾病 | 高血压 - acompany_with -> 糖尿病 |

### Neo4j 节点类型
| 节点类型 | 说明 | 主要属性 |
|---------|------|---------|
| `Disease` | 疾病 | name, desc, cause, prevent, easyGet, cureLasttime, curedProb |
| `Drug` | 药物 | name |
| `Department` | 科室 | name |
| `Symptom` | 症状 | name |
| `Check` | 检查项目 | name |

// api/employee.js
const BASE_URL = 'http://localhost:8080';

class EmployeeAPI {
  static async fetchEmployees(pageNum = 1, pageSize = 10, params = {}) {
    const url = new URL(`${BASE_URL}/admin/employee/page`);
    url.searchParams.append('pageNum', pageNum);
    url.searchParams.append('pageSize', pageSize);
    
    if (params.name) url.searchParams.append('name', params.name);
    if (params.department) url.searchParams.append('department', params.department);
    if (params.status) url.searchParams.append('status', params.status);
    
    return this.request(url);
  }

  static async getEmployeeById(id) {
    return this.request(`${BASE_URL}/admin/employee/${id}`);
  }

  static async addEmployee(data) {
    return this.request(`${BASE_URL}/admin/employee`, {
      method: 'POST',
      body: JSON.stringify(data)
    });
  }

  static async updateEmployee(data) {
    return this.request(`${BASE_URL}/admin/employee`, {
      method: 'PUT',
      body: JSON.stringify(data)
    });
  }

  static async deleteEmployee(id) {
    return this.request(`${BASE_URL}/admin/employee/${id}`, {
      method: 'DELETE'
    });
  }

  static async updatePerformance(id, performanceData) {
    const url = new URL(`${BASE_URL}/admin/employee/${id}/performance`);
    
    if (performanceData.performanceScore) {
      url.searchParams.append('performanceScore', performanceData.performanceScore);
    }
    if (performanceData.salesAmount) {
      url.searchParams.append('salesAmount', performanceData.salesAmount);
    }
    if (performanceData.completedOrders) {
      url.searchParams.append('completedOrders', performanceData.completedOrders);
    }
    
    return this.request(url, { method: 'PUT' });
  }

  static async getStatistics() {
    return this.request(`${BASE_URL}/admin/employee/statistics`);
  }

  static async request(url, options = {}) {
    const defaultOptions = {
      headers: {
        'Content-Type': 'application/json',
        'Authorization': 'Bearer your_token'
      }
    };
    
    const response = await fetch(url, { ...defaultOptions, ...options });
    const result = await response.json();
    
    if (result.code !== 200) {
      throw new Error(result.message);
    }
    
    return result.data;
  }
}

// 使用示例
const employees = await EmployeeAPI.fetchEmployees(1, 10, { name: '张三' });
const statistics = await EmployeeAPI.getStatistics();
